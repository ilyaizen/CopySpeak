  <name>Task 2: Fix IntersectionObserver for tab-aware sections</name>
  <files>src/lib/components/settings-page.svelte</files>
  <action>
    Update the IntersectionObserver to only observe sections for the active tab.

    **Replace setupScrollObserver function:**
    ```typescript
    function setupScrollObserver() {
      if (observer) observer.disconnect();

      observer = new IntersectionObserver(
        (entries) => {
          if (isManualScroll) return;
          // Find topmost visible section
          let topEntry: IntersectionObserverEntry | null = null;
          for (const entry of entries) {
            if (entry.isIntersecting) {
              if (!topEntry || entry.boundingClientRect.top < topEntry.boundingClientRect.top) {
                topEntry = entry;
              }
            }
          }
          if (topEntry) {
            const id = topEntry.target.id.replace("section-", "");
            if (activeSection !== id) activeSection = id;
          }
        },
        { rootMargin: "-120px0px -60% 0px", threshold: 0 }
      );

      // Only observe sections for the current tab
      const sections = tabSections[activeTab] || [];
      for (const section of sections) {
        const el = document.getElementById(`section-${section.id}`);
        if (el) observer.observe(el);
      }
    }
    ```

    **Add effect to re-setup observer when tab changes:**
    ```typescript
    $effect(() => {
      // Re-setup observer when tab changes
      if (activeTab) {
        requestAnimationFrame(() => {
          setupScrollObserver();
        });
      }
    });
    ```

    This ensures:
    - When user switches tabs, the observer disconnects from old sections
    - Observer reconnects to new tab's sections
    - Scroll highlighting works within each tab independently

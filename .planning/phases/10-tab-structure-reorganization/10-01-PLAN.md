  <name>Task 1: Create tab registry and activeTab state</name>
  <files>src/lib/components/settings-page.svelte</files>
  <action>
    Replace the flat `settingsCategories` array with a two-level tab/section registry.

    **Add new state:**
    ```typescript
    let activeTab = $state<"general" | "advanced" | "about">("general");
    ```

    **Replace settingsCategories with:**
    ```typescript
    const tabs = [
      { id: "general" as const, labelKey: "settings.tabs.general" },
      { id: "advanced" as const, labelKey: "settings.tabs.advanced" },
      { id: "about" as const, labelKey: "settings.tabs.about" }
    ];

    const tabSections: Record<string, { id: string; categoryKey: string; labelKey: string }[]> = {
      general: [
        { id: "app", categoryKey: "general", labelKey: "settings.sections.startup" },
        { id: "appearance", categoryKey: "appearance", labelKey: "settings.sections.appearance" },
        { id: "playback", categoryKey: "playback", labelKey: "settings.sections.playback" },
        { id: "hud", categoryKey: "hud", labelKey: "settings.sections.hud" },
        { id: "history", categoryKey: "history", labelKey: "settings.sections.history" },
        { id: "triggers", categoryKey: "triggers", labelKey: "settings.sections.triggers" },
        { id: "hotkeys", categoryKey: "hotkeys", labelKey: "settings.sections.hotkeys" }
      ],
      advanced: [
        { id: "pagination", categoryKey: "pagination", labelKey: "settings.sections.pagination" },
        { id: "sanitization", categoryKey: "sanitization", labelKey: "settings.sections.sanitization" }
      ],
      about: [
        { id: "about", categoryKey: "about", labelKey: "settings.sections.appInfo" }
      ]
    };
    ```

    **Add switchTab function:**
    ```typescript
    function switchTab(tabId: "general" | "advanced" | "about") {
      activeTab = tabId;
      const firstSection = tabSections[tabId][0];
      if (firstSection) {
        activeSection = firstSection.id;
      }
      // Reset scroll position for smooth UX
      window.scrollTo({ top: 0, behavior: "instant" });
    }
    ```

    Keep the existing `scrollToSection` function unchanged for now.

  </action>

  </verify>
  <done>
    - tabs array exists with 3 entries: general, advanced, about
    - tabSections registry maps each tab to its sections
    - activeTab state initialized to "general"
    - switchTab function sets activeTab and first section
  </done>

<task type="auto">
  <name>Task 2: Update sidebar to show tabs and sections</name>
  <files>src/lib/components/settings-page.svelte</files>
  <action>
    Refactor the sidebar navigation to show tabs at the top level, with sections shown below for the active tab.

    **Replace current sidebar nav (around line 215-228) with:**
    ```svelte
    <!-- Left Sidebar Menu (sticky) -->
    <aside class="w-28 shrink-0">
      <!-- Tab Navigation -->
      <nav class="sticky top-24">
        <div class="mb-2 space-y-0.5">
          {#each tabs as tab}
            <button
              class="w-full rounded-md px-2 py-1.5 text-left text-sm font-medium transition-colors {activeTab === tab.id
                ? 'bg-primary/10 text-primary border-l-2 border-primary'
                : 'text-muted-foreground hover:text-foreground hover:bg-muted/50'}"
              onclick={() => switchTab(tab.id)}
            >
              {$_(tab.labelKey)}
            </button>
          {/each}
        </div>

        <!-- Section Navigation (for active tab) -->
        {#if tabSections[activeTab].length > 0}
          <div class="mt-2 border-t border-border pt-2 space-y-0.5">
            {#each tabSections[activeTab] as section}
              <button
                class="w-full rounded-md px-2 py-1.5 text-left text-sm transition-colors {activeSection === section.id
                  ? 'bg-primary/10 text-primary border-l-2 border-primary'
                  : 'text-muted-foreground hover:text-foreground hover:bg-muted/50'}"
                onclick={() => scrollToSection(section.id)}
              >
                {$_(section.labelKey)}
              </button>
            {/each}
          </div>
        {/if}
      </nav>
    </aside>
    ```

    **Styling requirements:**
    - Tabs use font-medium weight (primary navigation)
    - Sections use normal weight (secondary navigation)
    - Active states use `bg-primary/10 text-primary border-l-2 border-primary`
    - Inactive states use `text-muted-foreground hover:text-foreground hover:bg-muted/50`
    - Divider line between tabs and sections when sections exist

  </action>
